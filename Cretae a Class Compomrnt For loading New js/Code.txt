In global.js file you can add js compoment 


class LoadComponent extends HTMLElement {
  constructor() {
    super();

     var element = this.querySelector(".media_slider");
     if (element) {
       $(element).slick({
       slidesToShow: 1,
  arrows: true,
  draggable: true,  
  dots: true,   
  autoplay: false,
  infinite: true,
  autoplaySpeed: 4000,
  pauseOnHover: false,
  pauseOnFocus: false  
        });
    }
  }
}
customElements.define('load-component', LoadComponent);
